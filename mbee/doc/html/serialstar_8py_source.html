139  def toHex(self, s):
140  lst = []
141  for ch in s:
142  hv = hex(ch).replace('0x', '')
143  if len(hv) == 1:
144  hv = '0' + hv
145  lst.append(hv)
146  return functools.reduce(lambda x, y: x + y, lst)

154  def read_frame_infos(self):
155  wait = True
156  self.bytes = []
157  length = 0
158  type_hex = "00"
159  while True:
160  start_byte = self.ser.read()
161  if(start_byte == b""):
162  return {"length" : 0, "frame_type" : "00"}
163  if self.toHex(start_byte).upper() == self.START_BYTE:
164  self.bytes.append(start_byte)
165  i = 1
166  detect = False
167  while i < 3:
168  cc = self.ser.read()
169  if(cc == b""):
170  return {"length" : 0, "frame_type" : "00"}
171  self.bytes.append(cc)
172  length_hex = ""
173  if self.toHex(cc).upper() in self.ESCAPED_CHARACTER:
174  detect = True
175  continue
176  else:
177  if detect:
178  length_hex += self.xor_hex(self.toHex(cc),"20")
179  detect = False
180  else:
181  length_hex += self.toHex(cc)
182  i += 1
183  length = int(length_hex, 16)
184  type_byte = self.ser.read()
185  if(type_byte == b""):
186  return {"length" : 0, "frame_type" : "00"}
187  self.bytes.append(type_byte)
188  type_hex = self.toHex(type_byte).upper()
189  return {"length" : length , "frame_type" : type_hex}

191  def read_rx_api(self, type = []):
192  while True:
193  infos = self.read_frame_infos()
194  length = infos["length"]
195  if (infos["frame_type"] == type or type == []) and length != 0:
196  while length > 0:
197  byte = self.ser.read()
198  if(byte == b""):
199  return "ERROR : received less than length bytes"
200  self.bytes.append(byte)
201  if not (self.toHex(byte).upper() in self.ESCAPED_CHARACTER):
202  length -= 1
203  return self.bytes
204  else:
205  return "ERROR : received a Wrong frame !!! "

207  def filter_frame(self):
208  if self.toHex(self.bytes[0]).upper() != self.START_BYTE:
209  return "ERROR invalid Frame"
210  self.fields = [self.START_BYTE]
211  detect = False
212  for j in range(1, len(self.bytes)):
213  byte = self.toHex(self.bytes[j]).upper()
214  if byte in self.ESCAPED_CHARACTER:
215  detect = True
216  else:
217  if detect :
218  self.fields.append(self.sub_hex2(byte,"20")[2:4])
219  detect=False
220  else:
221  self.fields.append(byte)
222  return self.fields

259  def send_local_at(self, frame_type, frame_id, at_command, at_parameter = ""):
260  if len(at_command) != 2:
261  return "Error : verify the Command AT !!!!"
262  else:
263  frame_local_at = []
264  ll = 4 + len(at_parameter) // 2
265  length = str(format(ll,"04x"))
266  frame_local_at.append(length[0:2])
267  frame_local_at.append(length[2:4])
268  frame_local_at.append(frame_type)
269  frame_local_at.append(frame_id)
270  frame_local_at.append(self.toHex(bytes(at_command.upper(), encoding = "ascii"))[0:2])
271  frame_local_at.append(self.toHex(bytes(at_command.upper(), encoding = "ascii"))[2:4])
272  if len(at_parameter) > 0:
273  for i in range(0, len(at_parameter), 2):
274  pp = at_parameter.upper()[i:i + 2]
275  frame_local_at.append(pp)
276  checksum = "00"
277  for i in range(2, len(frame_local_at)): #При подсчете контрольной суммы байты с длиной фрейма не учитываются.
278  checksum = self.add_hex2(frame_local_at[i], checksum)
279  checksum = (self.sub_hex2("FF", checksum[len(checksum) - 2 : len(checksum)])[2:4]).zfill(2)
280  frame_local_at.append(checksum)
281  frame_local_at.insert(0, self.START_BYTE) #SerialStar не поддерживает режимы с escape-символами в направлении хост->модуль, поэтому анализ на их наличие не призводится.
282  #print(frame_local_at)
283  tx_packet = binascii.unhexlify("".join(map(str, frame_local_at)))
284  self.ser.write(tx_packet)
285  time.sleep(self.API_FRAME_TX_TIMEOUT)
